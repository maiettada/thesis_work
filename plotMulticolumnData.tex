\documentclass{standalone}

\usepackage{tikz}
\usepackage{pgfplots}

\pgfplotsset{compat = newest}
\begin{document}
    \pgfplotstableread{multiple_functions.dat}{\table}
    \begin{tikzpicture}
        \begin{axis}[
            xmin = 0, xmax = 100,
            ymin = 0, ymax = 1,
            xtick distance = 10,
            ytick distance = 0.25,
            grid = both,
            minor tick num = 1,
            major grid style = {lightgray},
            minor grid style = {lightgray!25},
            width = \textwidth,
            height = 0.75\textwidth,
            legend cell align = {left},
            legend pos = north west
        ]
            \addplot[blue, mark = *] table [x = {x}, y = {f}] {\table};
            \addplot[red, only marks] table [x ={x}, y = {p}] {\table};
            \addplot[teal, only marks, mark = x, mark size = 3pt]table [x = {x}, y = {r}] {\table};
            \legend{
                F-score value,
                Precision,
                Recall
                }
        \end{axis}
    \end{tikzpicture}
\end{document}
